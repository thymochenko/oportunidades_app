SELECT ninja.nome_ninja as n_nome,ninja.id as id,p.nome as p_nome,ninja.idade as idade,c.nome as cla_nome,org.nome as org_nome,cat.classe as cat_nome,groups.nome as group_nome,ninja.velocidade as velocidade,ninja.forca as forca,ninja.inteligencia as intelig,ninja.energia as energia,ninja.ninjutsu as ninjutsu FROM ninja LEFT JOIN pais as p ON ninja.pais_id = p.id LEFT JOIN cla as c ON ninja.cla_id = c.id LEFT JOIN organizacao as org ON org.id = ninja.organizacao_id LEFT JOIN categoria as cat ON cat.id = ninja.categoria_id LEFT JOIN grupos as groups ON groups.id = ninja.grupos_id LEFT JOIN pais as p ON ninja.pais_id = p.id LEFT JOIN cla as c ON ninja.cla_id = c.id LEFT JOIN organizacao as org ON org.id = ninja.organizacao_id LEFT JOIN categoria as cat ON cat.id = ninja.categoria_id LEFT JOIN grupos as groups ON groups.id = ninja.grupos_id ORDER BY ninja.nome_ninja desc LIMIT 0,5
