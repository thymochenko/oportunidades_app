SELECT at.nome as at_nome,ninja.nome_ninja as n_nome,nivel_nomeclatura,ninja.id as id,p.nome as p_nome,ninja.idade as idade,at.elemento_id as el_id,at.nivel as nivel,el.nome as el_nome,at.id as ataque_id,ninja.velocidade as velocidade,ninja.forca as forca,ninja.inteligencia as intelig,ninja.ninjutsu as ninjutsu FROM ninja LEFT JOIN ataques_ninja as atN ON atN.ninja_id = ninja.id LEFT JOIN ataques as at ON at.id = atN.ataques_id LEFT JOIN elemento as el ON el.id = at.elemento_id LEFT JOIN pais as p ON ninja.pais_id = p.id LEFT JOIN ataques_ninja as atN ON atN.ninja_id = ninja.id LEFT JOIN ataques as at ON at.id = atN.ataques_id LEFT JOIN elemento as el ON el.id = at.elemento_id LEFT JOIN pais as p ON ninja.pais_id = p.id WHERE (atN.ninja_id = :id) ORDER BY at.nome desc LIMIT 100
